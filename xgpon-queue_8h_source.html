383 inline uint32_t
384 XgponQueue::CalculatePacketSize4Scheduling (uint32_t pktSizeInByte)
385 {
386  uint32_t sizeInByte = XgponXgemFrame::GetPaddedPayloadSize(pktSizeInByte); //after padding
387  sizeInByte += XgponXgemHeader::XGPON_XGEM_HEADER_LENGTH; //Xgem header overhead in word
388 
389  return (sizeInByte / 4);

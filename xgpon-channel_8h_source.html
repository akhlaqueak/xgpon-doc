130 inline uint16_t
131 XgponChannel::AddOnu (const Ptr<PonNetDevice>& device)
132 {
133  NS_ASSERT_MSG((m_onuDevices.size() < MAXIMAL_NODES_PER_XGPON), "Too many ONUs are attached to this channel.");
134 
135  m_onuDevices.push_back (device);
136 
137  //The delay is initialized to 0. It will be set later through "SetOnuPropagationDelay".
138  m_onuPropDelays.push_back (0);
139 
140  return m_onuDevices.size () - 1;
141 }
